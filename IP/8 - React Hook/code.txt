useEffect:
CounterOne.js code:
import React, { useState, useEffect } from 'react';

function CounterOne() {
  const [count, setCount] = useState(0);
  useEffect(() => {
    document.title = `You clicked ${count} times`;
  });

  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
    </div>
  )
  }
export default CounterOne
